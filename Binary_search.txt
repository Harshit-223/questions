#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cout << "size :";
    cin >> n;
    int arr[n];
    cout << "enter the elements : " ;
    for(int i =0;i<n;++i){
        cin >> arr[i];
    }
    int key;
    cout << "Enter the key : ";
    cin >> key ;
    int s = 0;
    int e = n-1;
    if(arr[s]<arr[e]){
        while(s<=e){
        int mid = s + (e-s)/2 ;
        if(arr[mid]==key){
            cout << mid;
            break;
        }
        if(arr[mid]<key){
            s = mid+1;
        }
        else{
            e = mid-1;
        }
    }
}
    else{
        while(s<=e){
        int mid = s + (e-s)/2 ;
        if(arr[mid]==key){
            cout << mid;
            break;
        }
        if(arr[mid]>key){
            s = mid+1;
        }
        else{
            e = mid-1;
        }
    }
        
    }
    
    
    return 0;
}
